2. run wav2mp3.exe
3. subfolder trash


# Required environment to run      
pip install pyinstaller    
pip install glob2  
pip install pydub  
pip install parmap  
pip install Send2Trash  

# run python
python wav2mp3.py


# pyinstaller build  
pyinstaller --onefile main.py  -n wav2mp3
